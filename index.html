#include<iostream>
using namespace std;

int main()
{
    int size;
    cin >> size;
    int ina[size];
    int ona[size];
    for(int i=0;i<size;i++){
        cin >> ina[i];
    }
    int pba[10];
    for(int i=0;i<10;i++){
        cin >> pba[i];
    }
    for(int i=0;i<size;i++){
        for(int j=0;j<10;j++){
            if(ina[i]==j)ona[i]=pba[j];
        }
    }
    for(int i=0;i<size;i++)cout << ona[i] << " ";
    return 0;
}
